# Cloner le dépot depuis GIT
git clone git@github.com:cledou/Cahier-de-labo.git

# vérifer que nodejs est installé
sudo apt install nodejs
# nodejs est déjà la version la plus récente
# se placer dans le répertoire du Cahier
cd Cahier-de-labo/www
# supprimer l'ancienne version des packages s'il elle existe
sudo rm package-lock.json

# compiler les packages
sudo npm install

#prendre un café, thé ou autre en attendant le message de fin (6mn sur mon Rockpi4)
# added 430 packages, and audited 431 packages in 6m
# 72 packages are looking for funding
# run `npm fund` for details

# vérifier que le programme se lance correctement (ctrl-c pour terminer)
sudo node Cahier

# vérifier que le serveur est opérationel
# lancer dans un navigateur internet le port 3006 (mettre l'adresse sur le réseau de votre machine)
http://192.168.1.xx:3006/

# installer le serveur comme service du système
# récupérer le chemin absolu du programme
sed 's,$PWD,'"$PWD"/',g' --in-place cahier.service
sudo mv cahier.service /etc/systemd/system/
sudo systemctl enable cahier.service
# sudo systemctl daemon-reload
